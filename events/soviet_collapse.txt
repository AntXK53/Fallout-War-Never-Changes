			# UPA
			every_state = {
				limit = {
					is_controlled_by = SOV

					has_state_flag = south_ukrainian_rebelions_sf
				}
				set_demilitarized_zone = no
				transfer_state_to = UPA
			}
			# RoU
			every_state = {
				limit = {
					is_controlled_by = SOV
					is_core_of = UKR
					NOT = {
						is_core_of = UPA
					}
				}
				transfer_state_to = UKR
			}
			# MOL
			MOL = {
				transfer_state = 78
			}
			# next country

				id = soviet_collapse.5

			}
		}
	}
